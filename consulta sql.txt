Reporte 1

SELECT cliente.CedulaOCe, CONCAT_WS('',cliente.PrimerNombre, cliente.PrimerApellido) as Nombre, cliente.DiasEnMora, if (cliente.DiasEnMora < 20 , 'Riesgo bajo', IF (cliente.DiasEnMora <= 35 , 'Riesgo medio','Riesgo alto') ) as riesgo , ciudad.Nombre as ciudad FROM cliente, ciudad ORDER BY `DiasEnMora` DESC


Reporte 2

SELECT sucursal.Nombre as sucursal, round(cotizacion.valorPendientePorPagar - cotizacion.ValorPolizaIvaIncl, 0) AS valorTotalPagado from sucursal, cotizacion ORDER BY `valorTotalPagado` DESC
